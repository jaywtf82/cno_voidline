<script>
  const card = document.getElementById('phase1-card');
  const dlg = document.getElementById('phase1-modal');
  const closeRed = document.getElementById('closeRed');

  function openModal(){ if (typeof dlg.showModal==='function') dlg.showModal(); else dlg.setAttribute('open',''); }
  function closeModal(){ if (typeof dlg.close==='function') dlg.close(); else dlg.removeAttribute('open'); }

  card.addEventListener('click', openModal);
  card.addEventListener('keydown', (e)=>{ if(e.key==='Enter' || e.key===' '){ e.preventDefault(); openModal(); } });

  closeRed.addEventListener('click', closeModal);
  dlg.addEventListener('click', (ev)=>{
    const box = dlg.querySelector('.modal').getBoundingClientRect();
    if(ev.clientX<box.left || ev.clientX>box.right || ev.clientY<box.top || ev.clientY>box.bottom){ closeModal(); }
  });
  document.addEventListener('keydown', (e)=>{ if(e.key==='Escape' && dlg.open){ closeModal(); }});
</script>
